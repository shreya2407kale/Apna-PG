// Firestore Security Rules for Apna PG
// Deploy these rules to Firebase Console > Firestore Database > Rules

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // ==================== USERS COLLECTION ====================
    match /users/{userId} {
      // Users can read their own data
      allow read: if request.auth.uid == userId;
      
      // Users can create their account during signup
      allow create: if request.auth.uid == userId && 
                       request.resource.data.keys().hasAll(['name', 'email', 'role']) &&
                       request.resource.data.role in ['User', 'Owner'];
      
      // Users can update their own profile
      allow update: if request.auth.uid == userId;
      
      // Users cannot delete their account
      allow delete: if false;
    }

    // ==================== PGS COLLECTION ====================
    match /pgs/{pgId} {
      // Anyone can read PG listings
      allow read: if request.auth != null;
      
      // Owners can create PGs
      allow create: if request.auth != null &&
                       get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'Owner' &&
                       request.resource.data.keys().hasAll(['title', 'ownerId', 'genderAllowed']) &&
                       request.resource.data.ownerId == request.auth.uid;
      
      // Owners can only update their own PGs
      allow update: if request.auth != null &&
                       resource.data.ownerId == request.auth.uid &&
                       request.resource.data.ownerId == resource.data.ownerId;
      
      // Owners can delete their own PGs
      allow delete: if request.auth != null &&
                       resource.data.ownerId == request.auth.uid;

      // ==================== ROOMS SUBCOLLECTION ====================
      match /rooms/{roomId} {
        allow read: if request.auth != null;
        
        allow create: if request.auth != null &&
                         get(/databases/$(database)/documents/pgs/$(pgId)).data.ownerId == request.auth.uid;
        
        allow update: if request.auth != null &&
                         get(/databases/$(database)/documents/pgs/$(pgId)).data.ownerId == request.auth.uid;
        
        allow delete: if request.auth != null &&
                         get(/databases/$(database)/documents/pgs/$(pgId)).data.ownerId == request.auth.uid;
      }
    }

    // ==================== BOOKINGS COLLECTION ====================
    match /bookings/{bookingId} {
      // Users can read their own bookings
      allow read: if request.auth.uid == resource.data.userId ||
                     request.auth.uid == resource.data.ownerId;
      
      // Users can create bookings
      allow create: if request.auth != null &&
                       request.auth.uid == request.resource.data.userId &&
                       request.resource.data.keys().hasAll(['userId', 'pgId', 'roomId', 'status']) &&
                       request.resource.data.status == 'pending';
      
      // Users can only update their own bookings (cancel)
      // Owners can update booking status
      allow update: if (request.auth.uid == resource.data.userId && 
                        request.resource.data.status == 'cancelled') ||
                       (request.auth.uid == resource.data.ownerId &&
                        request.resource.data.status in ['approved', 'rejected']);
      
      // Bookings cannot be deleted
      allow delete: if false;

      // Reason: All booking changes must be tracked via status updates
      // Keep booking history for disputes and analytics
    }

    // ==================== REVIEWS COLLECTION ====================
    match /reviews/{reviewId} {
      // Anyone can read reviews
      allow read: if request.auth != null;
      
      // Users can create reviews only if they have a completed booking
      allow create: if request.auth != null &&
                       request.auth.uid == request.resource.data.userId &&
                       request.resource.data.keys().hasAll(['userId', 'pgId', 'rating', 'comment']) &&
                       request.resource.data.rating >= 1 && 
                       request.resource.data.rating <= 5 &&
                       // Check if user has completed booking at this PG
                       exists(/databases/$(database)/documents/bookings/$(request.resource.data.bookingId));
      
      // Users can only update their own reviews
      allow update: if request.auth.uid == resource.data.userId &&
                       request.auth.uid == request.resource.data.userId;
      
      // Users can only delete their own reviews within 30 days
      allow delete: if request.auth.uid == resource.data.userId;
    }

    // ==================== DEFAULT RULES ====================
    // Deny all access by default
    match /{document=**} {
      allow read, write: if false;
    }
  }
}

// ==================== NOTES ====================
/*
SECURITY FEATURES IMPLEMENTED:

1. AUTHENTICATION
   - All operations require Firebase Authentication
   - User identity verified via request.auth.uid

2. ROLE-BASED ACCESS
   - Owner role check using user document
   - Different permissions for Users and Owners

3. DATA INTEGRITY
   - Required fields validated on create
   - Status values restricted to allowed values
   - Gender restrictions can be enforced in app logic

4. PRIVACY
   - Users only see their own bookings
   - Owners only see their PG bookings
   - Reviews visible to all authenticated users

5. BOOKING SAFETY
   - Bookings cannot be deleted (audit trail)
   - Only status can be changed
   - User can only cancel, owner can approve/reject

6. REVIEW INTEGRITY
   - Reviews require valid booking history
   - Only creator can update/delete
   - Rating range enforced (1-5)

7. PG OWNERSHIP
   - Owners can't change pgId after creation
   - Only PG owner can modify rooms
   - Can't transfer PG ownership

LIMITATIONS OF FIRESTORE RULES:
- No complex queries (e.g., "count bookings")
- No real-time checks for booking completion before review
  → Implement this check in frontend/backend
- No automatic index creation (Firebase will prompt)
- No transactions across documents

RECOMMENDED INDEX:
Create composite index for:
- bookings: userId + createdAt (descending)
- bookings: ownerId + createdAt (descending)
- pgs: ownerId + createdAt (descending)
- reviews: pgId + createdAt (descending)

Firebase Console will auto-suggest these or create them on first query.
*/
